## 5. Component Logic (Pseudo-Code)

```typescript
export function PhoneLookupFlow({
  branchId,
  onVisitorFound,
  onBack,
  onRegisterNew, // New prop
  className,
}: PhoneLookupFlowProps): React.ReactElement {
  const [state, setState] = React.useState({
    phoneValue: '',
    lookupState: 'idle',
    visitorData: null,
  });

  const phoneInputRef = React.useRef<HTMLInputElement>(null);

  /**
   * Handles register new visitor
   * Updated to invoke callback if provided
   */
  const handleRegisterNew = React.useCallback((): void => {
    onRegisterNew?.(state.phoneValue);
  }, [state.phoneValue, onRegisterNew]);

  /**
   * Handles search another - resets to initial state
   */
  const handleSearchAnother = React.useCallback((): void => {
    setState({
      phoneValue: '',
      lookupState: 'idle',
      visitorData: null,
    });
    setTimeout(() => {
      phoneInputRef.current?.focus();
    }, 100);
  }, []);

  // Render VisitorNotFoundDisplay when lookupState is 'not_found'
  return (
    <div className={cn('space-y-6', className)}>
      {/* Existing phone input and search UI */}

      {state.lookupState === 'not_found' && (
        <Card>
          <CardContent className="pt-6">
            <div className="text-center space-y-4">
              <div className="mx-auto w-12 h-12 rounded-full bg-red-100 flex items-center justify-center">
                <AlertCircle className="h-6 w-6 text-red-600" />
              </div>

              <div className="space-y-2">
                <h3 className="text-lg font-semibold">Visitor Not Found</h3>
                <p className="text-sm text-gray-600">
                  We couldn't find a visitor with this number.
                </p>
              </div>

              <div className="bg-gray-50 rounded-lg p-3">
                <p className="text-sm font-medium text-gray-700">{state.phoneValue}</p>
              </div>

              <div className="space-y-3 pt-2">
                <Button
                  onClick={handleRegisterNew}
                  className="w-full"
                  size="lg"
                  data-testid="register-new-button"
                >
                  <UserPlus className="mr-2 h-5 w-5" />
                  Register as new visitor
                </Button>

                <Button
                  onClick={handleSearchAnother}
                  variant="ghost"
                  className="w-full"
                  data-testid="search-another-button"
                >
                  Search another
                </Button>
              </div>
            </div>
          </CardContent>
        </Card>
      )}
    </div>
  );
}
```
